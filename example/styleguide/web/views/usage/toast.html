@Injectable()
class AppController {
    final _logger = new Logger('wsk_toast.AppController');

    int _messageCounter = 1;
    final WskToast wskToast;

    AppController(this.wskToast) {
    }

    void showSimpleToast() {
        wskToast("Message #${_messageCounter++}").show()
                    .then( (final WskDialogStatus status) {

            _logger.info("Status: $status");
        });
    }
}

class SampleModule extends Module {
    SampleModule() {
        install(new WskToastModule());
    }
}

main() {
    configLogger();
    applicationFactory().addModule(new SampleModule()).rootContextType(AppController).run();
}
                
<wsk-button ng-click="showSimpleToast()">
    Show simple
</wsk-button>

As you can see the top / left or top / right position regards the page-header.
This is very easy to configure:
main.scss:

body > .wsk-toast-container {
    .wsk-toast {
        &.wsk-toast--top {
            top: $layout-header-desktop-row-height;
            @media screen and (max-width: $layout-screen-size-threshold) {
                top: $layout-header-mobile-row-height;
            }
        }
    }
}
