import 'package:wsk_angular/wsk_panel/wsk_panel.dart';

@Injectable()
class AppController {
    AppController() {
    }
}

class SampleModule extends Module {
    SampleModule() {
        install(new WskPanelModule());
        bind(AppController);
    }
}

main() {
    applicationFactory()
        .addModule(new SampleModule())
        .rootContextType(AppController).run();
}
