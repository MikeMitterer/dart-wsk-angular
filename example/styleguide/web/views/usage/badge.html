import 'package:wsk_angular/wsk_badge/wsk_badge.dart';

@Injectable()
class AppController {
    AppController() {
    }
}

class SampleModule extends Module {
    SampleModule() {
        install(new WskBadgeModule());
        bind(AppController);
    }
}

main() {
    applicationFactory()
        .addModule(new SampleModule())
        .rootContextType(AppController).run();
}
